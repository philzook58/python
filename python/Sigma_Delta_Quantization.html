T = 10.0
N = 10000
t = np.linspace(0,T,N)

def plotspec(y):
    plt.plot(np.fft.fftfreq(N,T/N)[:100] , np.log(0.001+np.abs(sp.fft(y)[:100])))

def plotfullspec(y):
    plt.plot(np.fft.fftfreq(N,T/N) , np.log(0.001+np.abs(sp.fft(y))))
    
def bitquantize(y):
    return np.sign(y)
